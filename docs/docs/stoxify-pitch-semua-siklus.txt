// src/lib/api.ts
const BASE = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000/api/v1'

export async function apiFetch<T>(path: string, opts?: RequestInit): Promise<T> {
  const res = await fetch(BASE + path, {
    ...opts,
    headers: { 'Content-Type': 'application/json', ...opts?.headers },
    credentials: 'include',  // untuk httpOnly cookie refresh token
  })
  if (!res.ok) {
    const err = await res.json()
    throw new Error(err?.error?.message || "API error")
  }
  return res.json()
}

# app/modules/auth/service.py
import bcrypt, secrets
from app.core.redis import redis_client

async def register_user(data: UserCreate, db) -> dict:
    existing = await db.fetchrow("SELECT id FROM users WHERE email=$1", data.email)
    if existing:
        raise HTTPException(409, detail={'code':'EMAIL_EXISTS'})

    hashed = bcrypt.hashpw(data.password.encode(), bcrypt.gensalt(12))
    user = await db.fetchrow(
        "INSERT INTO users(email,phone_wa,name,hashed_password) VALUES($1,$2,$3,$4) RETURNING id",
        data.email, data.phone_wa, data.name, hashed.decode()
    )

    # OTP: 6 digit, simpan di Redis 5 menit
    otp = str(secrets.randbelow(900000) + 100000)
    await redis_client.setex(f"otp:{user['id']}:register", 300, otp)

    if data.channel == 'wa':
        await send_wa_otp(data.phone_wa, otp)
    else:
        await send_email_otp(data.email, otp)

    return {'user_id': str(user['id']), 'message': f'OTP dikirim ke {data.channel}'}

Auth Guard Middleware (Nasywa)

// src/middleware.ts
import { NextRequest, NextResponse } from 'next/server'

const PUBLIC = ['/', '/login', '/register', '/pricing', '/verify-otp', '/forgot-password']

export function middleware(req: NextRequest) {
  const isPublic = PUBLIC.some(p => req.nextUrl.pathname.startsWith(p))
  const token = req.cookies.get('stx_access_token')?.value
  if (!isPublic && !token)
    return NextResponse.redirect(new URL('/login', req.url))
  return NextResponse.next()
}
export const config = { matcher: ['/((?!api|_next/static|_next/image|favicon).*)'] }

'use client'
import { useEffect, useRef } from 'react'
import { createChart, ColorType } from 'lightweight-charts'

export function PriceChart({ symbol, exchange, timeframe }) {
  const ref = useRef(null)
  useEffect(() => {
    if (!ref.current) return
    const chart = createChart(ref.current, {
      layout: { background: { type: ColorType.Solid, color: '#010206' }, textColor: '#FFFEF8' },
      grid: { vertLines: { color: 'rgba(255,254,248,0.05)' }, horzLines: { color: 'rgba(255,254,248,0.05)' } },
      width: ref.current.clientWidth, height: 400,
    })
    const series = chart.addCandlestickSeries({
      upColor: '#4ADE80', downColor: '#F87171',
      borderUpColor: '#4ADE80', borderDownColor: '#F87171',
      wickUpColor: '#4ADE80', wickDownColor: '#F87171',
    })
    fetchChartData(symbol, exchange, timeframe).then(d =>
      series.setData(d.candles.map(c => ({ time: c.time, open: +c.open, high: +c.high, low: +c.low, close: +c.close })))
    )
    return () => chart.remove()
  }, [symbol, exchange, timeframe])
  return <div ref={ref} className="w-full" />
}

Kalkulasi P&L Holdings (Fahmi)

# app/modules/portfolio/service.py
from decimal import Decimal

async def calculate_holdings(portfolio_id: str, db, redis) -> list:
    assets = await db.fetch("SELECT * FROM portfolio_assets WHERE portfolio_id=$1", portfolio_id)
    result = []
    total_value = Decimal("0")

    for a in assets:
        price_data = await redis.hgetall(f"price:{a['symbol']}:{a['exchange']}")
        current    = Decimal(price_data.get("price", "0"))
        qty        = Decimal(str(a["quantity"]))
        avg_buy    = Decimal(str(a["avg_buy_price"]))

        current_value = current * qty
        pnl           = current_value - (avg_buy * qty)
        pnl_pct       = (pnl / (avg_buy * qty) * 100) if avg_buy > 0 else Decimal("0")
        total_value  += current_value

        result.append({
            "symbol":        a["symbol"],
            "current_price": str(current),
            "pnl":           str(pnl.quantize(Decimal("0.00"))),
            "pnl_pct":       str(pnl_pct.quantize(Decimal("0.00"))),
        })

    for h in result:
        w = Decimal(h["pnl"]) / total_value * 100 if total_value else Decimal("0")
        h["weight_pct"] = str(w.quantize(Decimal("0.00")))
    return result

WEIGHTS = { "technical": 0.25, "fundamental": 0.25, "sentiment": 0.25, "ml": 0.25 }
LABELS = [(80,"strong_buy"), (60,"buy"), (40,"neutral"), (20,"sell"), (0,"strong_sell")]

async def calculate_ai_score(symbol: str, exchange: str) -> dict:
    tech  = await get_technical_score(symbol, exchange)
    fund  = await get_fundamental_score(symbol, exchange)
    sent  = await get_sentiment_score(symbol, exchange)
    ml    = await get_ml_score(symbol, exchange)

    score = int(tech*0.25 + fund*0.25 + sent*0.25 + ml*0.25)
    label = next(l for threshold, l in LABELS if score >= threshold)

    return {
        "score":         score,
        "label":         label,
        "is_gold_badge": score >= 90,
        "breakdown":     { "technical":tech, "fundamental":fund, "sentiment":sent, "ml":ml }
    }

# app/modules/intraday/service.py
from decimal import Decimal

def calculate_trade_plan(data) -> dict:
    entry  = Decimal(data.entry_price)
    target = Decimal(data.target_price)
    sl     = Decimal(data.stop_loss)
    capital= Decimal(data.capital)
    risk   = Decimal(data.risk_per_trade_pct) / 100

    max_loss      = capital * risk
    risk_per_share= abs(entry - sl)
    lots          = int(max_loss / risk_per_share / 100) if risk_per_share > 0 else 0
    shares        = lots * 100
    pos_size      = entry * shares
    profit        = abs(target - entry) * shares
    actual_loss   = risk_per_share * shares
    rr            = profit / actual_loss if actual_loss > 0 else Decimal("0")
    fee           = pos_size * Decimal("0.002")  # 0.2% avg fee IDX

    TWO = Decimal("0.00")
    return {
        "recommended_lots":     lots,
        "position_size":        str(pos_size.quantize(TWO)),
        "max_loss":             str(actual_loss.quantize(TWO)),
        "potential_profit":     str(profit.quantize(TWO)),
        "risk_reward":          str(rr.quantize(TWO)),
        "fee_estimate":         str(fee.quantize(TWO)),
        "net_profit_after_fee": str((profit - fee).quantize(TWO)),
        "capital_used_pct":     str((pos_size/capital*100).quantize(TWO)),
    }

# app/modules/chat/websocket.py
from fastapi import WebSocket, WebSocketDisconnect
from groq import AsyncGroq
from app.core.redis import redis_client

groq_client = AsyncGroq()
DAILY_LIMITS = {"lite": 500, "growth": 2000, "apex": None}

async def chat_handler(ws: WebSocket, user, db):
    await ws.accept()
    session_id = await create_chat_session(user.id, db)

    try:
        while True:
            msg = await ws.receive_json()
            if msg["type"] != "message": continue

            limit = DAILY_LIMITS.get(user.tier)
            if limit:
                used = int(await redis_client.get(f"tokens:{user.id}:{today()}") or 0)
                if used >= limit:
                    await ws.send_json({"type":"error","code":"TOKEN_LIMIT_REACHED"})
                    continue

            stream = await groq_client.chat.completions.create(
                model="llama-3.3-70b-versatile",
                messages=build_messages(msg["content"], user),
                stream=True, max_tokens=500
            )

            total = 0
            async for chunk in stream:
                delta = chunk.choices[0].delta.content or ""
                if delta:
                    await ws.send_json({"type":"chunk","content":delta,"session_id":session_id})
                if chunk.usage: total = chunk.usage.total_tokens

            await redis_client.incrby(f"tokens:{user.id}:{today()}", total)
            await redis_client.expire(f"tokens:{user.id}:{today()}", 86400)
            remaining = (limit - total) if limit else None
            await ws.send_json({"type":"done","tokens_used":total,"tokens_remaining":remaining})

    except WebSocketDisconnect:
        await close_chat_session(session_id, db)

# app/core/dependencies.py

ROLE_ORDER = {
    "moderator":1, "content":2, "cs":3,
    "marketing":4, "finance":5, "admin":6, "super_admin":7
}

def require_admin_role(min_role: str):
    async def check(user=Depends(get_current_user), db=Depends(get_db)):
        admin = await db.fetchrow(
            "SELECT role FROM admin_users WHERE user_id=$1 AND is_active=TRUE", user.id
        )
        if not admin:
            raise HTTPException(403, detail={"code":"NOT_ADMIN"})
        if ROLE_ORDER.get(admin["role"], 0) < ROLE_ORDER.get(min_role, 999):
            raise HTTPException(403, detail={"code":"INSUFFICIENT_ROLE"})
        return user
    return check

// src/components/ui/ErrorBoundary.tsx
'use client'
import { Component, ErrorInfo, ReactNode } from 'react'

interface Props { children: ReactNode; fallback?: ReactNode }
interface State { hasError: boolean }

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false }
  static getDerivedStateFromError(): State { return { hasError: true } }
  componentDidCatch(error: Error, info: ErrorInfo) {
    console.error("ErrorBoundary:", error, info)
  }
  render() {
    if (this.state.hasError) {
      return this.props.fallback || (
        <div className="flex items-center justify-center h-40">
          <div className="text-center">
            <p className="text-[#F87171] font-[Syne] font-bold mb-2">Terjadi kesalahan</p>
            <button onClick={() => this.setState({ hasError: false })}
              className="text-white/50 text-sm underline">Coba lagi</button>
          </div>
        </div>
      )
    }
    return this.props.children
  }
}
